% Предикат для получения подарков, которые Мэри подарила Тому
podarila_mary_tomu(X) :- podaril(mary, tom, X).

podarili_tomu(X) :- findall(P, (podaril(_, tom, P); podaril(_, tom, P)), TempList), list_to_set(TempList, X).
