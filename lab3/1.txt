% Правило, которое определяет места (1, 2 или 3)
место(1).
место(2).
место(3).

% Правило, которое устанавливает ограничения на места Гриши и Толи
ограничение_Гриши(X) :- место(X), X \== 2, X \== 3.
ограничение_Толи(X) :- место(X), X \== 3.

% Основное правило для определения мест каждого ребенка
определить_места(Юра, Гриша, Толя) :-
    место(Юра), место(Гриша), место(Толя),
    ограничение_Гриши(Гриша),
    ограничение_Толи(Толя),
    Юра \== Гриша, Юра \== Толя, Гриша \== Толя.

% Запрос для поиска решения
запрос_1 :- определить_места(Юра, Гриша, Толя), write("Решение: "), write(Юра), write(" "), write(Гриша), write(" "), write(Толя), nl.

% Запуск запроса
main :-
    запрос_1.

% Запуск программы
:- main.
